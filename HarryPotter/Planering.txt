        {
            Gryffindor gryffindor = new Gryffindor();

            Ravenclaw ravenclaw = new Ravenclaw();
            Hufflepuff hufflepuff = new Hufflepuff();

            if (gryffindor.Password == PasswordBox.Text)
            {
                MessageBox.Show("Welcome to Gryffindor!");
            }
            else if (slytherin.Password == PasswordBox.Text)
            {
                MessageBox.Show("Welcome to Slytherin!");
            }

            {
                MessageBox.Show("Welcome to Ravenclaw!");
